55 void UKUISubContainer::SetTotalSize( float fWidth, float fHeight )

100 void UKUISubContainer::UpdateRenderCacheSize()
101 {
102  if ( !IsRenderCaching() )
103  return;
104 
105  FVector2D v2Size = GetSize();
106  const FVector2D v2TotalSize = GetTotalSize();
107 
108  v2Size.X = min( v2Size.X, v2TotalSize.X );
109  v2Size.Y = min( v2Size.Y, v2TotalSize.Y );
110 
111  if ( v2Size.X > 0.9f && v2Size.Y > 0.9f )
112  {
113  oRenderCache->SetSizeStruct( v2Size );
114  oRenderCache->SetTextureCoordsStruct( GetCornerOffset() / v2TotalSize );
115  oRenderCache->SetTextureSizeStruct( ( v2Size - GetCornerOffset() ) / v2TotalSize );
116  }
117 }

132 void UKUISubContainer::Render( AKUIInterface* aHud, UCanvas* oCanvas, const FVector2D& v2Origin, UKUIInterfaceElement* oRenderCacheObject )
133 {
134  if ( oContainerFor != NULL )
135  SetTotalSizeStruct( oContainerFor->GetSize() + oContainerFor->GetMarginSize() );
136 
137  if ( !HasValidLayout() )
138  DoLayout();
139 
140  if ( IsRenderCaching() )
141  {
142  // Redraw the contents (using the canvas and origin supplied by the render cache.)
143  if ( oRenderCacheObject == this )
144  {
145  oCanvas->Reset();
146 
147  KUISendEvent( FKUIInterfaceElementRenderEvent, EKUIInterfaceElementEventList::E_Render, oCanvas, v2Origin );
148 
149  for ( int32 i = 0; i < arChildren.Num(); ++i )
150  {
151  if ( arChildren[ i ] == NULL )
152  continue;
153 
154  if ( !arChildren[ i ]->IsVisible() )
155  continue;
156 
157  arChildren[ i ]->Render( aHud, oCanvas, FVector2D::ZeroVector, oRenderCacheObject );
158  }
159  }
160 
161  // Draw the cache to the output
162  else
163  {
164  if ( !HasValidAlignLocation() )
165  {
166  TArray<UKUIInterfaceElement*> arAlignStack;
167 
168  CalculateAlignLocation( arAlignStack );
169  }
170 
171  if ( !oRenderCache->IsRenderCacheValid() )
172  oRenderCache->UpdateRenderCache( this );
173 
174  v2LastScreenRenderLocation = v2Origin + GetRenderLocation();
175  oCanvas->Reset();
176  oRenderCache->Render( aHud, oCanvas, v2LastScreenRenderLocation, oRenderCacheObject );
177 
178  KUISendEvent( FKUIInterfaceElementRenderEvent, EKUIInterfaceElementEventList::E_Render, oCanvas, v2Origin );
179  }
180  }
181 }

225 bool UKUISubContainer::IsMouseOver() const
226 {
227  if ( v2LastScreenRenderLocation.X == -1.f )
228  return false;
229 
230  if ( GetInterface() == NULL )
231  {
232  KUIErrorUO( "Null interface" );
233  return false;
234  }
235 
236  if ( GetContainer() != NULL )
237  {
238  if ( !GetContainer()->IsMouseOver() )
239  return false;
240  }
241 
242  const FVector2D v2CursorLocation = GetInterface()->GetCursorLocation();
243  const FVector2D v2Size = GetSize();
244  const FVector2D v2TopLeftLocation = GetScreenLocation() + v2CornerOffset;
245 
246  if ( v2CursorLocation.X >= v2TopLeftLocation.X &&
247  v2CursorLocation.Y >= v2TopLeftLocation.Y &&
248  v2CursorLocation.X < ( v2TopLeftLocation.X + v2Size.X ) &&
249  v2CursorLocation.Y < ( v2TopLeftLocation.Y + v2Size.Y ) )
250  return true;
251 
252  return false;
253 }
