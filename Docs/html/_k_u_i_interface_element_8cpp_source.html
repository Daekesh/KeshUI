537 void UKUIInterfaceElement::AddAlignedToThis( UKUIInterfaceElement* oAlignChild )
538 {
539  // Can't remove a null pointer.
540  if ( oAlignChild == NULL )
541  {
542  KUIErrorUO( "Trying to add a null align child" );
543  return;
544  }
545 
546  if ( oAlignChild == this )
547  {
548  KUIErrorUO( "Trying to align to self" );
549  return;
550  }
551 
552  if ( oAlignChild->GetAlignedTo() != this )
553  {
554  KUIErrorUO( "Trying to add an align child that isn't aligned to this" );
555  return;
556  }
557 
558  // Check if we're in the child array already.
559  for ( int32 i = 0; i < arAlignedToThis.Num(); ++i )
560  if ( arAlignedToThis[ i ] == oAlignChild )
561  return;
562 
563  // If this child already has a container, remove it from it.
564  if ( oAlignChild->GetAlignedTo() != NULL )
565  oAlignChild->GetAlignedTo()->RemoveAlignedToThis( oAlignChild );
566 
567  // Add the child to our array.
568  arAlignedToThis.Add( oAlignChild );
569 }

572 void UKUIInterfaceElement::RemoveAlignedToThis( UKUIInterfaceElement* oAlignChild )
573 {
574  // Can't remove a null pointer
575  if ( oAlignChild == NULL )
576  {
577  KUIErrorUO( "Trying to remove null align child" );
578  return;
579  }
580 
581  // Check if we're actually in the children array.
582  int32 iIndex = INDEX_NONE;
583 
584  for ( int32 i = 0; i < arAlignedToThis.Num(); ++i )
585  {
586  if ( arAlignedToThis[ i ] != oAlignChild )
587  continue;
588 
589  iIndex = i;
590  break;
591  }
592 
593  // It wasn't
594  if ( iIndex == INDEX_NONE )
595  {
596  KUIErrorDebugUO( "Trying to remove an align child not aligned to this element" );
597  return;
598  }
599 
600  // Swap out the highest index with this child, to maintain contiguity.
601  int32 iHighestIndex = INDEX_NONE;
602 
603  for ( int32 i = arAlignedToThis.Num() - 1; i > iIndex; --i )
604  {
605  if ( !arAlignedToThis[ i ].IsValid() )
606  continue;
607 
608  iHighestIndex = i;
609  break;
610  }
611 
612  // We found a highest child, so swap. Otherwise, we were the highest child and no sort is necessary.
613  if ( iHighestIndex != INDEX_NONE )
614  arAlignedToThis[ iIndex ] = arAlignedToThis[ iHighestIndex ];
615 
616  // Reduce the size of the array by 1.
617  arAlignedToThis.SetNum( iIndex );
618 }

621 void UKUIInterfaceElement::CalculateAlignLocation( TArray<UKUIInterfaceElement*>& arAlignStack )
622 {
623  if ( HasValidAlignLocation() )
624  return;
625 
626  // Avoid recursive layout dependencies.
627  if ( arAlignStack.Contains( this ) )
628  return;
629 
630  if ( GetContainer() != NULL && GetContainer()->IsChildsLayoutManaged( this ) )
631  {
632  //SetAlignLocation( FVector2D::ZeroVector );
633 
634  // Caused circular layout dependencies.
635  //if ( !GetContainer()->HasValidLayout() )
636  //GetContainer()->DoLayout();
637 
638  return;
639  }
640 
641  UKUIInterfaceElement* oAlignedTo = GetAlignedTo();
642 
643  if ( oAlignedTo == NULL )
644  oAlignedTo = GetContainer();
645 
646  if ( oAlignedTo == NULL )
647  {
648  SetAlignLocation( FVector2D::ZeroVector );
649  return;
650  }
651 
652  arAlignStack.Add( this );
653 
654  if ( !oAlignedTo->HasValidAlignLocation() )
655  oAlignedTo->CalculateAlignLocation( arAlignStack );
656 
657  const FVector2D v2Size = GetSize() + GetMarginSize();
658  FVector2D v2AlignLocation = FVector2D::ZeroVector;
659  FVector2D v2AlignSize = FVector2D::ZeroVector;
660  FVector2D v2AlignRenderLocation = FVector2D::ZeroVector;
661 
662  // If we're aligned to something in this container, use local space
663  if ( oAlignedTo->GetContainer() == GetContainer() )
664  {
665  v2AlignRenderLocation = oAlignedTo->GetRenderLocation() - oAlignedTo->GetMarginOffset();
666  v2AlignSize = oAlignedTo->GetSize() + oAlignedTo->GetMarginSize();
667  }
668 
669  // If we're aligned to something outside our container, use its screen location
670  else if ( oAlignedTo != GetContainer() )
671  {
672  v2AlignRenderLocation = oAlignedTo->GetScreenLocation() - oAlignedTo->GetMarginOffset();
673  v2AlignSize = oAlignedTo->GetSize() + oAlignedTo->GetMarginSize();
674  }
675 
676  // We're aligned to the container, so use 0, 0
677  else
678  {
679  v2AlignRenderLocation = FVector2D::ZeroVector;
680  v2AlignSize = oAlignedTo->GetSize();
681  }
682 
683  v2AlignLocation = CalculateAlignLocation(
684  v2AlignRenderLocation,
685  v2AlignSize,
686  v2Size
687  );
688 
689  // If we're aligned to something outside of our container, move the align location back into local space.
690  if ( oAlignedTo != GetContainer() && oAlignedTo->GetContainer() != GetContainer() )
691  v2AlignLocation += oAlignedTo->GetScreenLocation() - GetContainer()->GetScreenLocation();
692 
693  arAlignStack.Remove( this );
694  SetAlignLocation( v2AlignLocation );
695 }

698 const FVector2D UKUIInterfaceElement::CalculateAlignLocation( const FVector2D& v2Origin, const FVector2D& v2Extent, const FVector2D& v2Size, const FVector2D& v2Default )
699 {
700  FVector2D v2AlignLocation;
701  const FVector2D v2MarginSize = GetMarginSize(); // X+Z, Y+W
702 
703  if ( bDebug )
704  KUILogUO( "(%f,%f) (%f,%f) (%f,%f) (%f,%f)", ExpandV2( v2Origin ), ExpandV2( v2Extent ), ExpandV2( v2Size ), ExpandV2( v2Default ) );
705 
706  switch ( GetHorizontalAlignment() )
707  {
708  case EKUIInterfaceHAlign::HA_Left:
709  v2AlignLocation.X = v2Origin.X;
710  break;
711 
712  case EKUIInterfaceHAlign::HA_Centre:
713  v2AlignLocation.X = v2Origin.X + ( v2Extent.X / 2.f ) - ( v2Size.X / 2.f );
714  break;
715 
716  case EKUIInterfaceHAlign::HA_Right:
717  v2AlignLocation.X = v2Origin.X + v2Extent.X - v2Size.X;
718  break;
719 
720  case EKUIInterfaceHAlign::HA_Left_Outer:
721  v2AlignLocation.X = v2Origin.X - v2Size.X;
722  break;
723 
724  case EKUIInterfaceHAlign::HA_Right_Outer:
725  v2AlignLocation.X = v2Origin.X + v2Extent.X;
726  break;
727 
728  case EKUIInterfaceHAlign::HA_Fill:
729  SetSize( v2Extent.X - v2MarginSize.X, GetSize().Y );
730  v2AlignLocation.X = v2Origin.X;
731  break;
732 
733  case EKUIInterfaceHAlign::HA_None:
734  default:
735  v2AlignLocation.X = v2Default.X;
736  break;
737  }
738 
739  switch ( GetVerticalAlignment() )
740  {
741  case EKUIInterfaceVAlign::VA_Top:
742  v2AlignLocation.Y = v2Origin.Y;
743  break;
744 
745  case EKUIInterfaceVAlign::VA_Centre:
746  v2AlignLocation.Y = v2Origin.Y + ( v2Extent.Y / 2.f ) - ( v2Size.Y / 2.f );
747  break;
748 
749  case EKUIInterfaceVAlign::VA_Bottom:
750  v2AlignLocation.Y = v2Origin.Y + v2Extent.Y - v2Size.Y;
751  break;
752 
753  case EKUIInterfaceVAlign::VA_Top_Outer:
754  v2AlignLocation.Y = v2Origin.Y - v2Size.Y;
755  break;
756 
757  case EKUIInterfaceVAlign::VA_Bottom_Outer:
758  v2AlignLocation.Y = v2Origin.Y + v2Extent.Y;
759  break;
760 
761  case EKUIInterfaceVAlign::VA_Fill:
762  SetSize( GetSize().X, v2Extent.Y - v2MarginSize.Y );
763  v2AlignLocation.Y = v2Origin.Y;
764  break;
765
