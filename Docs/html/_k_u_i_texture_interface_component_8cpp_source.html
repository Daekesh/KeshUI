28 void UKUITextureInterfaceComponent::SetTexture( UTexture* tTexture )
29 {
30  if ( this->tTexture == tTexture )
31  return;
32 
33  bool bSetSize = true;
34 
35  if ( this->tTexture != NULL )
36  bSetSize = ( this->tTexture->GetSurfaceWidth() == GetSize().X && this->tTexture->GetSurfaceHeight() == GetSize().Y );
37 
38  else
39  bSetSize = ( GetSize().X == 0.f && GetSize().Y == 0.f );
40 
41  this->tTexture = tTexture;
42 
43  Invalidate();
44 
45  if ( bSetSize )
46  {
47  if ( this->tTexture != NULL )
48  SetSize( this->tTexture->GetSurfaceWidth(), this->tTexture->GetSurfaceHeight() );
49 
50  else
51  SetSize( 0.f, 0.f );
52  }
53 }
