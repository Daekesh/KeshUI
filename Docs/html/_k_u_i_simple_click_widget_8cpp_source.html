51 bool UKUISimpleClickWidget::OnMouseButtonDown( const FKUIInterfaceContainerMouseButtonEvent& stEventInfo )
52 {
53  if ( stEventInfo.bHandled )
54  return true;
55 
56  if ( !CanReceieveMouseEvents() )
57  return Super::OnMouseButtonDown( stEventInfo );
58 
59  if ( stEventInfo.eButton != EMouseButtons::Left )
60  return Super::OnMouseButtonDown( stEventInfo );
61 
62  bDown = false;
63 
64  if ( !bMouseOver )
65  return Super::OnMouseButtonDown( stEventInfo );
66 
67  if ( !IsVisibleRecursive() )
68  return Super::OnMouseButtonDown( stEventInfo );
69 
70  if ( IsDisabledRecursive() )
71  return Super::OnMouseButtonDown( stEventInfo );
72 
73  bDown = true;
74  v2ClickOffset = stEventInfo.v2Location - GetScreenLocation();
75  KUISendSubEvent( FKUIInterfaceEvent, EKUIInterfaceWidgetEventList::E_StateChange );
76 
77  return true;

81 bool UKUISimpleClickWidget::OnMouseButtonUp( const FKUIInterfaceContainerMouseButtonEvent& stEventInfo )
82 {
83  if ( stEventInfo.bHandled )
84  return true;
85 
86  if ( !CanReceieveMouseEvents() )
87  return Super::OnMouseButtonUp( stEventInfo );
88 
89  if ( stEventInfo.eButton != EMouseButtons::Left )
90  return Super::OnMouseButtonUp( stEventInfo );
91 
92  if ( !bDown )
93  return Super::OnMouseButtonUp( stEventInfo );
94 
95  bool bTriggerClick = ( IsVisibleRecursive() && !IsDisabledRecursive() );
96 
97  if ( bTriggerClick && bMouseOver )
98  {
99  FKUIInterfaceWidgetClickEvent stClickEventInfo( EKUIInterfaceWidgetEventList::E_Click, v2ClickOffset );
100  SendEvent( stClickEventInfo );
101  }
102 
103  bDown = false;
104  FKUIInterfaceEvent stStateEventInfo( EKUIInterfaceWidgetEventList::E_StateChange );
105  SendEvent( stStateEventInfo );
106 
107  if ( bTriggerClick )
108  return bMouseOver;
109 
110  return Super::OnMouseButtonUp( stEventInfo );
111 }
